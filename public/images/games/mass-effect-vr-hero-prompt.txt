import os, sys, json, requests
os.chdir('/home/antforce/.hermes/hermes-agent')
if os.path.exists('.env'):
    with open('.env') as f:
        for line in f:
            if '=' in line and not line.strip().startswith('#'):
                key, value = line.strip().split('=', 1)
                os.environ[key] = value.strip().strip('"').strip("'")
sys.path.insert(0, '.')
from tools.image_generation_tool import image_generate_tool

Towering curved architecture stretching into atmospheric distance, deep navy #0b1020 atmospheric background with ethereal cyan #6ee7ff holographic energy streams. if r.get('success'):
    url = r['image']
    resp = requests.get(url, timeout=90)
    with open('/home/antforce/compoundvr-project/public/images/games/mass-effect-vr-hero.jpg', 'wb') as f:
        f.write(resp.content)
